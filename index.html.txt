// -------------------------------------- TAB ----------------------------------
function showTab(t){
  document.querySelectorAll('.panel').forEach(p=>p.classList.remove('active'));
  document.querySelectorAll('.tab').forEach(b=>b.classList.remove('active'));
  document.getElementById('p-'+t).classList.add('active');
  event.currentTarget.classList.add('active');
}

// -------------------------------------- REPORT LOGIC --------------------------
let repRunning = false;
let repTimer = null;
let repCount = 0;
let successCount = 0;
let failCount = 0;
let repLimit = 100;
let targetUID = '';

// Helper to log to terminal
function repLog(msg, cls='tt'){
  const l = document.getElementById('rep-log');
  l.innerHTML += `\n<span class="${cls}">${msg}</span>`;
  l.scrollTop = l.scrollHeight;
}

// Update target preview dynamically
function updateMockTarget() {
  const uid = document.getElementById('rep-url').value;
  document.getElementById('mock-uid').textContent = uid || 'N/A';
}

function repStart(){
  targetUID = document.getElementById('rep-url').value;
  if(!targetUID) {
    repLog('[!] Vui lňng nh?p Target UID/Link', 'tr');
    return;
  }
  
  repLimit = parseInt(document.getElementById('rep-limit').value) || 100;
  
  // UI Update
  document.getElementById('rep-start').disabled = true;
  document.getElementById('rep-stop').disabled = false;
  repRunning = true;
  repCount = 0;
  successCount = 0;
  failCount = 0;
  
  document.getElementById('mock-status').className = 'status-badge';
  document.getElementById('mock-status').textContent = 'Tr?ng thái: Đang b? t?n công...';
  
  repLog(`[*] Kh?i t?o chi?n d?ch Report...`, 'ta');
  repLog(`[*] Target: ${targetUID}`, 'ta');
  repLog(`[*] Limit: ${repLimit} reports`, 'ta');
  repLog(`[*] Loading accounts & proxies...`, 'tm');
  
  // Fake delay for startup
  setTimeout(() => {
    repLog(`[+] S?n sŕng! B?t d?u d?y d? li?u...`, 'tg');
    processReport();
  }, 1000);
}

function processReport(){
  if(!repRunning) return;
  
  if(repCount >= repLimit){
    repLog(`\n[?] CHI?N D?CH HOŔN T?T. Đă g?i ${repLimit} requests.`, 'tg');
    repLog(`[!] Ch? ph?n h?i t? h? th?ng ki?m duy?t...`, 'ty');
    document.getElementById('mock-status').className = 'status-badge banned';
    document.getElementById('mock-status').textContent = 'Tr?ng thái: B? khóa t?m th?i (Checkpoint)';
    repStop();
    return;
  }

  repCount++;
  const delay = parseInt(document.getElementById('rep-delay').value) || 200;
  
  // Simulate success or fail (85% success rate for sending)
  const isSuccess = Math.random() > 0.15;
  
  if(isSuccess) {
    successCount++;
    if(repCount % 5 === 0) { // Log less frequently to avoid flooding
      repLog(`[?] Clone ID ${Math.floor(Math.random()*90000)+10000} -> Sent Report payload`, 'tg');
    }
  } else {
    failCount++;
    if(repCount % 5 === 0) {
      repLog(`[x] Proxy ${Math.floor(Math.random()*255)}.${Math.floor(Math.random()*255)}.x.x b? t? ch?i k?t n?i`, 'tr');
    }
  }

  // Update UI Stats
  document.getElementById('rep-s-sent').textContent = repCount;
  document.getElementById('rep-s-success').textContent = successCount;
  document.getElementById('rep-s-fail').textContent = failCount;
  
  const pct = Math.round((repCount / repLimit) * 100);
  document.getElementById('rep-pbar').style.width = pct + '%';
  document.getElementById('rep-pi-cur').textContent = `${repCount} / ${repLimit}`;
  document.getElementById('rep-pi-pct').textContent = pct + '%';

  // Next iteration
  repTimer = setTimeout(processReport, delay);
}

function repStop(){
  repRunning = false;
  clearTimeout(repTimer);
  document.getElementById('rep-start').disabled = false;
  document.getElementById('rep-stop').disabled = true;
  if(repCount > 0 && repCount < repLimit) {
    repLog(`[-] Đă d?ng th? công.`, 'ty');
  }
}

function repReset(){
  repStop();
  document.getElementById('rep-log').innerHTML = '<span class="tm">$ ready. waiting for command...</span>';

  document.getElementById('mock-status').className = 'status-badge';
  document.getElementById('mock-status').textContent = 'Tr?ng thái: Ho?t d?ng';
}
